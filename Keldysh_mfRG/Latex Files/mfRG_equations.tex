\subsection*{General properties of the four-point vertex}

\begin{align}
\Gamma = \Gamma_{1^\prime 2^\prime | 1 2} = \V\Gamma12121212{1010} \vqw12121212{1010}
\end{align}

\begin{itemize}
\item[(I)] 
frequency conservation: $\omega_1^\prime+\omega_2^\prime = \omega_1 + \omega_2$
\\
$\Rightarrow$ introduce new frequencies
\begin{align}
A &= \omega_2^\prime - \omega_1 = \omega_2 - \omega_1^\prime \\
\Pi &= \omega_1 + \omega_2 = \omega_1^\prime + \omega_2^\prime \\
T &= \omega_1^\prime - \omega_1 = \omega_2 - \omega_2^\prime
\end{align}
or reversely
\begin{align}
\omega_1 &= \tfrac12 (-A + \Pi -T ) \\
\omega_2 &= \tfrac12 (A + \Pi + T) \\
\omega_1^\prime &= \tfrac12 (-A + \Pi + T) \\
\omega_2^\prime &= \tfrac12 (A + \Pi - T)
\end{align}

\item[(II)]
spin conservation: $\sigma_1^\prime + \sigma_2^\prime = \sigma_1 + \sigma_2$
\begin{itemize}
\item[1)] all spins equal: $\Gamma_{\sigma \sigma | \sigma \sigma}$
\item[2)] incoming spins reversed: $\Gamma_{\sigma \bar{\sigma} | \sigma \bar{\sigma}}$ or $\Gamma_{\sigma \bar{\sigma} | \bar{\sigma} \sigma}$
\end{itemize}

\item[(III)]
particle exchange:
\begin{align}
& \V\Gamma21122112{1010} \vqw21122112{1010} & 
| & & & \V\Gamma21122112{1010} (q_2^\prime, q_1^\prime | q_1 q_2 \,,\, T, \Pi, A) \\
&= \V\Gamma12211221{1010} \vqw12211221{1010} & 
| & & & \V\Gamma21122112{1010} (q_1^\prime, q_2^\prime | q_2 q_1 \,,\, -T, \Pi, -A) \\
&= - \V\Gamma21212121{1010} \vqw21212121{1010} & 
| & & & \V\Gamma21122112{1010} (q_2^\prime, q_1^\prime | q_2 q_1 \,,\, -A, \Pi, -T) \\
&= - \V\Gamma12121212{1010} \vqw12121212{1010} & 
| & & & \V\Gamma21122112{1010} (q_1^\prime, q_2^\prime | q_1 q_2 \,,\, A, \Pi, T)
\end{align}

\item[(IV)]
complex conjugation:
\begin{align}

&= (-1)^{1 + \alpha_1 + \alpha_2 + \alpha_1^\prime + \alpha_2^\prime} \

\V\Gamma12121212{0101} (q_1 q_2 | q_1^\prime q_2^\prime \,,\, A, \Pi, -T)
\end{align}

\item[(V)]
causality: $\Gamma^{cc|cc} = 0$

\end{itemize}
