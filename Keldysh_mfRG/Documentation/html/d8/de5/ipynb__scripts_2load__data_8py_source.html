21def load_SE_A(filename, iLambda=0):
22 """
23 Load self-energy and spectral function from hdf5 file.
24
25 Parameters:
26 filename: name of the file from which data should be read
27 iLambda: Lambda iteration that should be read off (default: 0)
28
29 Returns:
30 v: vector of frequencies at which result is saved
31 SigmaR: retarded self-energy
32 SigmaK: Keldysh self-energy
33 A: spectral function
34 """
35 with h5py.File(filename, 'r') as f:
36 v = np.array(f["ffreqs"])
37 self = np.array(f["selflist"][iLambda])
38
39 _, _, _, epsilon, _, Gamma, U, _, Lambdas = load_parameters(filename)
40
41 nv = len(v)
42 Sigma = np.zeros((2, nv), dtype=complex)
43 for iK in range(2):
44 for iv in range(len(v)):
45 Sigma[iK][iv] = self[iK * nv + iv][0] + 1j * self[iK * nv + iv][1]
46
47 G = 1/(v - epsilon + (Gamma+Lambdas[iLambda])/2*1j - Sigma[0])
48 A = -np.imag(G)/np.pi
49
50 for iv in range(len(v)):
51 Sigma[0][iv] -= U/2.
52
53 return v, Sigma, A

55def load_K1(filename, iLambda=0):
56 """
57 Load K1 class from hdf5 file.
58
59 Parameters:
60 filename: name of the file from which data should be read
61 iLambda: Lambda iteration that should be read off (default: 0)
62
63 Returns:
64 w: vector of frequencies at which result is saved
65 K1a: K1 in the a channel
66 K1p: K1 in the p channel
67 K1t: K1 in the t channel
68 """
69 with h5py.File(filename, 'r') as f:
70 w = np.array(f["bfreqs"])
71 K1_a = np.array(f["K1_a"][iLambda])
72 K1_p = np.array(f["K1_p"][iLambda])
73 K1_t = np.array(f["K1_t"][iLambda])
74
75 nw = len(w)
76 K1a = np.zeros((2, nw), dtype=complex)
77 K1p = np.zeros((2, nw), dtype=complex)
78 K1t = np.zeros((2, nw), dtype=complex)
79 for iK in range(2):
80 for iw in range(nw):
81 K1a[iK][iw] = K1_a[iK * nw + iw][0] + 1j * K1_a[iK * nw + iw][1]
82 K1p[iK][iw] = K1_p[iK * nw + iw][0] + 1j * K1_p[iK * nw + iw][1]
83 K1t[iK][iw] = K1_t[iK * nw + iw][0] + 1j * K1_t[iK * nw + iw][1]
84
85 return w, K1a, K1p, K1t

111def load_data(U_NRG, Delta_NRG, filenames):
112 """
113 Load all fRG data
114
115 Parameters:
116 U_NRG: value of the interaction U for NRG data at which to load fRG data
117 Delta_NRG: effective hybridization of NRG data (usually Gamma_NRG=1 ==> Delta_NRG=0.5)
118 filenames: list of file names from which the data should be obtained
119 """
120
121 DIAG_CLASS_list, N_LOOPS_list = [], []
122 T_list, epsilon_list, mu_list = [], [], []
123 Gamma_list, Lambda_list, Delta_list, U_list, V_list = [], [], [], [], []
124 v_list, Sigma_list, A_list = [], [], []

127
128 for filename in filenames:
129 # load parameters
130 DIAG_CLASS, N_LOOPS, T, epsilon, mu, Gamma, U, V, Lambdas = load_parameters(filename)
131
132 # determine effective hybridization Delta
133 Deltas = (Gamma + Lambdas) / 2
134
135 # check at which Lambda iteration U/Delta matches between fRG and the desired NRG value
136 if(U_NRG == 0):
137 iLambda = 0
138 else:
139 try:
140 iLambda = np.where(U/Deltas == U_NRG/Delta_NRG)[0][0]
141 except IndexError:
142 iLambda = np.argmin(np.abs(U/Deltas - U_NRG/Delta_NRG))
143 print("Actual U/Delta-value = " + str( (2*U)/(Lambdas[iLambda] + Gamma)))
144
145 # fix Lambda and effective hybridization Delta
146 Lambda = Lambdas[iLambda]
147 Delta = (Gamma + Lambda) / 2
148
149 # load self-energy and spectral function
150 v, Sigma, A = load_SE_A(filename, iLambda=iLambda)
151
152 # load K1
153 w, K1a, K1p, K1t = load_K1(filename, iLambda=iLambda)
154
155 # load K2
156 if DIAG_CLASS >= 2:
157 w2, v2, K2a, K2p, K2t = load_K2(filename, iLambda=iLambda)
158 else:
159 w2, v2, K2a, K2p, K2t = [], [], [], [], []
160
161 # add all data to lists
162 DIAG_CLASS_list.append(DIAG_CLASS)
163 N_LOOPS_list.append(N_LOOPS)
164 T_list.append(T)
165 epsilon_list.append(epsilon)
166 mu_list.append(mu)
