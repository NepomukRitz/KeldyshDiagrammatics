700 void update_grid(const frequencyGrid_type& frequencies_new, const this_class& buffer4data) {
701 using buffer_type = multidimensional::multiarray<Q,rank>;
702 buffer4data.initInterpolator();
703 const dimensions_type dims = base_class::get_dims();

706#pragma omp parallel

716 // interpolate old values to new vector
717 Q value = buffer4data.interpolate_impl(freqs, idx);
718 data_new.at(idx) = value;
719 }
720 base_class::set_vec(std::move(data_new)); // update vertex to new interpolated values
721 base_class::set_VertexFreqGrid(frequencies_new);
722 buffer4data.set_initializedInterpol(false);
