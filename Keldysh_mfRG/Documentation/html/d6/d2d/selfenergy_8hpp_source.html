201template <typename Q> auto SelfEnergy<Q>::valsmooth(int iK, double v, int i_in) const -> Q {//smoothly interpolates for values between discrete frequency values of mesh
202
203 if (std::abs(v) > Sigma.frequencies.get_wupper_b() + inter_tol) //Check the range of frequency. If too large, return Sigma(\infty)
204 //Returns asymptotic value (Hartree contribution for retarded and 0. for Keldysh component)
205 return (1.-(double)iK)*(this->asymp_val_R);
206 else {
207 Q result;
208#ifdef BARE_SE_FEEDBACK
209 frequencies_type freqs = {0.0};
210#else
211 frequencies_type freqs = {v};
212#endif
213 index_type idx;
214 idx[my_defs::SE::keldysh]= iK;
215 idx[my_defs::SE::internal]= i_in;
216 result = Sigma.interpolate_impl(freqs, idx);
217 return result + (1.-(double)iK)*(this->asymp_val_R);
218 }

347 if(p==0){ //max norm
