125template <typename Q, bool differentiated> void State<Q,differentiated>::initialize(bool checks) {
126 // Initial conditions
127 // Assign initial conditions to self energy
128 if (!KELDYSH && PARTICLE_HOLE_SYMMETRY) {
129 this->selfenergy.initialize(0., 0.);
130 }
131 else {
132 this->selfenergy.initialize(config.U / 2., 0.);
133 if (!PARTICLE_HOLE_SYMMETRY){ // SIAM in Keldysh WITHOUT particle-hole symmetry
134 assert (not PARTICLE_HOLE_SYMMETRY);
135 Hartree_Solver Hartree_Term(Lambda, config);
136 const double hartree_value = Hartree_Term.compute_Hartree_term_bracketing(1e-12, checks, checks);
137 this->selfenergy.initialize(hartree_value, 0.);
138 vertex.center_frequency_grids(shift_fgrids_woPHS ? hartree_value + config.epsilon : 0.);
139 }
140 }
141
142 // Assign initial conditions to bare vertex
143 if (KELDYSH and CONTOUR_BASIS != 1) this->vertex.initialize(-config.U/2.);
144 else this->vertex.initialize(-config.U);
145
146 this->initialized = true;
147}
