340 template<typename gridType>
341 auto shrink_freq_box_impl(const gridType &freqGrid, const double rel_tail_threshold,
342 const vec<double> &maxabs_along_x, const bool verbose) const -> gridType {
343 const size_t number_of_frequency_points_along_w = maxabs_along_x.size();
344 assert(freqGrid.get_all_frequencies().size() == maxabs_along_x.size());
345
346 gridType frequencies_new = freqGrid;
347
348 if (maxabs_along_x.max_norm() < 1e-20) return frequencies_new; // don't shrink if there is no data yet
349
350 int index = -1;
351 while (true) {
352 if (maxabs_along_x[number_of_frequency_points_along_w - 1 - (index + 1)] >= rel_tail_threshold) break;
353 index++;
354 }
355 if (index > -1) { // if the frequency box is too big, shrink to appropriate size
356 double t_belowthresh = freqGrid.get_auxiliary_gridpoint(
357 index); // auxiliary frequency point before passing threshold
358 double t_abovethresh = freqGrid.get_auxiliary_gridpoint(
359 index + 1); // auxiliary frequency point after passing threshold
360 double h = (rel_tail_threshold - maxabs_along_x[number_of_frequency_points_along_w - 1 - index]) * (t_abovethresh - t_belowthresh) /
361 (maxabs_along_x[number_of_frequency_points_along_w - 1 - (index + 1)] - maxabs_along_x[number_of_frequency_points_along_w - 1 - index]);
362 const double safety = 0.9;
363 frequencies_new.set_w_upper(std::abs(freqGrid.frequency_from_t(t_belowthresh + h * safety)));
364 } else if (index == -1) { // if data on outermost grid point is too big, then enlarge the box OR print warning
365 //double t_upper_new = 1 - maxmax*rel_tail_threshold * (1-freqGrid.t_upper) / (maxabs_along_x[maxabs_along_x.size() -1]);
366 //double w_upper_new = frequencies_new.frequency_from_t(t_upper_new);
367 //frequencies_new.set_w_upper(w_upper_new);
368 }
369 frequencies_new.initialize_grid();
370
371 return frequencies_new;
372 }
