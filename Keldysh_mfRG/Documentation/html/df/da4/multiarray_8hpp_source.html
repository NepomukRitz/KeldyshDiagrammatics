275 template <my_index_t numberFrequencyDims, my_index_t pos_first_freqpoint, my_index_t vecsize, my_index_t sample_size>
276 auto get_values(const index_type& index) const -> Eigen::Matrix<T, vecsize, numberFrequencyDims == 1 ? sample_size : (numberFrequencyDims == 2 ? sample_size*sample_size : sample_size*sample_size*sample_size)>
277 {
278#ifndef NDEBUG
279 assert(check_bounds(index));
280#endif
281 Eigen::Matrix<T, vecsize, numberFrequencyDims == 1 ? sample_size : (numberFrequencyDims == 2 ? sample_size*sample_size : sample_size*sample_size*sample_size)> result;
282 const size_t flat_ini = get_flatindex_ini<pos_first_freqpoint+numberFrequencyDims>(index);
283
284 if constexpr (numberFrequencyDims == 1) {
285 for (my_index_t i = 0; i < sample_size; i++) {
286 const auto res = at_vectorized<vecsize>(flat_ini + m_length_cumulative[pos_first_freqpoint]*i);
287 result.col(i) = res;
288 }
289 }
290 else if constexpr(numberFrequencyDims == 2) {
291 for (my_index_t i = 0; i < sample_size; i++) {
292 for (my_index_t j = 0; j < sample_size; j++) {
293 result.col(i * sample_size + j) = at_vectorized<vecsize>(flat_ini + m_length_cumulative[pos_first_freqpoint]*i + m_length_cumulative[pos_first_freqpoint+1]*j);
294
295 }
296 }
297 }
298 else if constexpr (numberFrequencyDims == 3) {
299 for (my_index_t i = 0; i < sample_size; i++) {
300 for (my_index_t j = 0; j < sample_size; j++) {
301 for (my_index_t l = 0; l < sample_size; l++) {
302 result.col(i * sample_size*sample_size + j * sample_size + l) = at_vectorized<vecsize>( flat_ini + m_length_cumulative[pos_first_freqpoint] * i + m_length_cumulative[pos_first_freqpoint + 1] * j + m_length_cumulative[pos_first_freqpoint + 2] * l);
303 }
304 }
305 }
306 }
307 else {
308 assert(false); // numberFrequencyDims > 3 not supported
309 }
310 return result;
311 }

763 // binary op for a multiarray and a scalar
764 template <
765 typename Q, typename L, typename R, size_t depth,
766 std::enable_if_t<!std::is_same_v<std::result_of_t<Q && (L &&, R &&)>, L>, bool> = true>
767 auto scalar_map(Q op, const multiarray<L, depth> &lhs, const R &rhs)
768 {
769 multiarray<std::result_of_t<Q && (L &&, R &&)>, depth> res(
770 lhs.length());
771 for (size_t i = 0; i < res.size(); i++)
