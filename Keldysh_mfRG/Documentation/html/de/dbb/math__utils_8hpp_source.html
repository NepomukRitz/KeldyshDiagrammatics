81template<int degreeplus, typename Q>
82inline auto lagrangePoly(const Q x, const double (&xs)[degreeplus], const Q (&ys) [degreeplus]) -> Q {
83 Q result = 0.;
84
85 double denominator, numerator;
86 for (int i = 0; i < degreeplus; i++) {
87 numerator = 1.;
88 denominator = 1.;
89 for (int k = 0; k < i; k++) {
90 denominator *= (xs[i] - xs[k]);
91 numerator *= (x - xs[k]);
92 }
93 for (int k = i+1; k < degreeplus; k++) {
94 denominator *= (xs[i] - xs[k]);
95 numerator *= (x - xs[k]);
96 }
97
98 result += ys[i] * numerator/denominator;
99 }
100
101 return result;
102}

246inline void getMultIndexSkippingOneDimension(std::array<size_t,rank>& indx, const size_t iflat, const std::array<size_t,rank>& dims)
247{
248 static_assert(idim < rank, "idim must be smaller than rank.");
249 size_t temp = iflat;
250 size_t dimtemp = 1;
251 for (unsigned int it = 1; it < idim; it++) {
252 dimtemp *= dims[it];

254 for (unsigned int it = idim+1; it < rank; it++) {
255 dimtemp *= dims[it];
256 }

258 indx[0] = temp / dimtemp;

262 dimtemp = dimtemp / dims[it];

264 temp -= indx[it] * dimtemp;

267 dimtemp = dimtemp / dims[it];

300 assert(i_dim <= rank);

504 template <typename T, size_t rank>
505 T get_finite_differences_helper(const vec<T>& data, const vec<freqType>& xs, const std::array<size_t,rank>& dims,
506 const std::array<size_t,rank>& permutation, size_t it_dimsum, size_t jt, vec<int>& no_i, int lower, int upper) {
507 T result = 0;
508 double prefactor_data0 = 0.;
509 for (int i : no_i) {
510 double numerator=1.;
511 double denominator = 1.;
512 for (int j = lower; j < i; j++) {
513 denominator *= (xs[jt + i] - xs[jt + j]);
514 assert(std::abs(denominator) > 1e-15);
515 }
516 for (int j =i+1; j <=upper; j++) {
517 denominator *= (xs[jt + i]-xs[jt + j]);
518 assert(std::abs(denominator) > 1e-15);
519 }
520 for (int j : no_i) numerator *= (xs[jt ]-xs[jt + j]);
521 numerator /= (xs[jt ]-xs[jt + i]);
522 result += numerator / denominator * data[rotateFlatIndex(it_dimsum + jt + i, dims, permutation)];
523
524
525 prefactor_data0 += 1./(xs[jt ]-xs[jt + i]);
526 }
527 result += prefactor_data0 * data[rotateFlatIndex(it_dimsum + jt, dims, permutation)];
528 return result;
529 }
