43def compile_stuff(cpp_target: string):
44 text = """
45#!/bin/bash
46
47
48# load necessary modules
49export CLUSTER=ASC
50module load cmake
51
52
53export LANG=C
54export LC_ALL=C
55
56dir_path=$(dirname "$(realpath "$0")") # directory of this script
57src_dir="$dir_path"/.. # source directory
58build_dir="$src_dir"/build # build directory
59mkdir "$build_dir"
60
61# navigate to the directory with the CMakeLists.txt file
62cd "$build_dir" || exit
63
64
65# load cmake project
66cmake -S$src_dir -B$build_dir
67
68
69# build
70TARGET=""" + cpp_target + """
71cmake --build . --target $TARGET
72
73mv ./$TARGET $src_dir
74
75GREEN='\033[1;32m' # green
76NC='\033[0m' # no color
77echo -e "\n Moved executable ${GREEN}$TARGET ${NC}to Keldysh_mfRG source directory.\n"
78 """
79 f = open("./compile_asc_saveIntegrand.sh", "w")
80 f.write(text)
81 close_hdf_file(f)
82 print("Compiling " + cpp_target)
83 os.system("bash ./compile_asc_saveIntegrand.sh")
84 os.system("rm ./compile_asc_saveIntegrand.sh")
