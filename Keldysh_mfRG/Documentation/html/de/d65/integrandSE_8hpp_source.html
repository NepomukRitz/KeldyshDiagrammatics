258 int npoints = 1e5;

266 wu = propagator.selfenergy.Sigma.frequencies.primary_grid.w_upper * 2.;

301auto IntegrandSE<Q,vertType,all_spins,return_type,version>::Keldysh_value(const double vp) const -> return_type {
302#if SWITCH_SUM_N_INTEGRAL
303
304 buffertype_propagator G1 = evaluate_propagator_vectorized( vp);
305 buffertype_propagator G2 = evaluate_propagator_vectorized(-vp);
306
307 buffertype_vertex V1 = evaluate_vertex_vectorized( vp);
308 buffertype_vertex V2 = evaluate_vertex_vectorized(-vp);
309
310if constexpr(std::is_same_v<return_type,double> or std::is_same_v<return_type,comp>) {
311 const return_type result = (G1*V1 + G2*V2).eval()[0] * 0.5;
312 return result;
313}
314else {
315 if constexpr(CONTOUR_BASIS) {
316 const return_type result = (G1*V1*transform_to_KeldyshBasis + G2*V2*transform_to_KeldyshBasis).eval() * 0.5;
317 return result;
318
319 }
320 else {
321 const return_type result = (G1*V1 + G2*V2).eval() * 0.5;
322 return result;
323 }
324}
325
326#else
327 Q Gi, Gi2;
328 evaluate_propagator(Gi , iK, vp);
329 evaluate_propagator(Gi2, iK,-vp);
330
331 Q factorClosedAbove, factorClosedAbove2;
332#ifdef SYMMETRIZED_SELF_ENERGY_FLOW
333 Q factorClosedBelow;
334 evaluate_vertex(factorClosedAbove, factorClosedBelow, iK, vp);
335 return (1./2.) * Gi * (factorClosedAbove + factorClosedBelow);
336#else
337 evaluate_vertex(factorClosedAbove, iK, vp);
338 evaluate_vertex(factorClosedAbove2, iK,-vp);
339 return (Gi * factorClosedAbove + Gi2 * factorClosedAbove2) * 0.5;
340#endif // SYMMETRIZED_SELF_ENERGY_FLOW
341#endif // SWITCH_SUM_N_INTEGRAL

345template<typename Q, typename vertType, bool all_spins, typename return_type, bool version>
346auto IntegrandSE<Q,vertType,all_spins,return_type,version>::Matsubara_value(const double vp) const -> Q {
347 Q GM;
348 evaluate_propagator(GM, vp);
349
350 Q factorClosedAbove = evaluate_vertex_vectorized(vp);
351
352 if (!PARTICLE_HOLE_SYMMETRY) {
353 return (GM * factorClosedAbove);
354 }
355 else {
356 // in the particle-hole symmetric_full case in Matsubara we only save the imaginary part of the selfenergy Im(Sigma)
357 // Accordingly the saved propagator is -Im(G)
358 // Hence we need an additional factor of -1
359 return -(GM * factorClosedAbove);
360 }
361}

364void IntegrandSE<Q,vertType,all_spins,return_type,version>::evaluate_propagator(Q &Gi, const int iK, const double vp) const {
365#if CONTOUR_BASIS != 1
366 switch (iK) {
367 case 0:
368 Gi = propagator.valsmooth(0, vp, i_in); // retarded propagator (full or single scale)
369 break;
370 case 1:
371 Gi = myconj(propagator.valsmooth(0, vp, i_in)); // advanced propagator (full or single scale)
372 break;
373 case 2:
374 Gi = propagator.valsmooth(1, vp, i_in); // Keldysh propagator (full or single scale)
375 break;
376 default:;
377 }
378#else
379 Gi = propagator.valsmooth(iK, vp, i_in);
380#endif
