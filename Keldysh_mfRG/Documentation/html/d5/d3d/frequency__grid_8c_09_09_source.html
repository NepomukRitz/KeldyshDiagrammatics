284double grid_transf_v1(const double w, const double W_scale, const double w_center) {
285 // Version 1: linear around w=0, good for w^(-2) tails
286 const double w_dev = w - w_center;
287 return w_dev/sqrt(W_scale*W_scale + w_dev*w_dev);
288}
289double grid_transf_inv_v1(const double t, const double W_scale, const double w_center) {
290 // Version 1: linear around w=0, good for w^(-2) tails
291 return w_center + W_scale*t/sqrt(1.-t*t);
292}
293double integration_measure_v1(const double t, const double W_scale) {
294 double temp = sqrt(1 - t*t);
295 return W_scale / (temp*temp*temp);
296}

299double grid_transf_v2(const double w, const double W_scale, const double w_center) {
300 // Version 2: quadratic around w=0, good for w^(-2) tails
301 const double w_dev = w - w_center;
302 const double w2 = w_dev * w_dev;
303 const double res = sgn(w_dev) * sqrt((sqrt(w2*w2 + 4 * w2 * W_scale * W_scale) - w2) / 2.) / W_scale;
304 assert(isfinite(res));
305 return res;
306}
307double grid_transf_inv_v2(const double t, const double W_scale, const double w_center) {
308 // Version 2: quadratic around w=0, good for w^(-2) tails
309 return w_center + W_scale * t * std::abs(t) / sqrt(1. - t * t);
310}
311double integration_measure_v2(const double t, const double W_scale) {
312 double temp = sqrt(1 - t*t);
313 return W_scale * sgn(t) * t * (2 - t*t) / (temp*temp*temp);
314}

322double grid_transf_v3(const double w, const double W_scale, const double w_center) {
323 // Version 3: linear around w=0, good for w^(-1) tails
324 const double almost_zero = 1e-12;
325 const double w_dev = w - w_center;
326 return (std::abs(w_dev) < almost_zero) ? 0. : (-W_scale + sqrt(4*w_dev*w_dev + W_scale*W_scale))/2/w_dev;
327}
328double grid_transf_inv_v3(const double t, const double W_scale, const double w_center) {
329 // Version 3: linear around w=0, good for w^(-1) tails
330 return w_center + W_scale * t / (1.-t*t);
331}
332double integration_measure_v3(const double t, const double W_scale) {
333 double temp = t*t;
334 return W_scale * (1 + temp) / (1 - temp) / (1 - temp);
335}

337double grid_transf_v4(const double w, const double W_scale, const double w_center) {
338 // Version 4: quadratic around w=0, good for w^(-1) tails
339 const double w_dev = w - w_center;
340 const double abs_w = std::abs(w_dev);
341 return sgn(w_dev) * sqrt(abs_w/(abs_w + W_scale));
342}
343double grid_transf_inv_v4(const double t, const double W_scale, const double w_center) {
344 // Version 4: quadratic around w=0, good for w^(-1) tails
345 return w_center + W_scale * sgn(t) * t*t /(1.-t*t);
346}
347double integration_measure_v4(const double t, const double W_scale) {
348 double temp = 1 - t*t;
349 return 2 * W_scale * sgn(t) * t / temp / temp;
350}

372freqType wscale_from_wmax_v1(freqType & Wscale, const freqType w1, const freqType wmax, const int N) {
373 freqType Wscale_candidate;
374
375 // Version 1: linear around w=0, good for w^(-2) tails
376 Wscale_candidate = wmax * sqrt( (N*N -1*2) / (pow(wmax/w1, 2) - N*N));
377 return std::max(Wscale, Wscale_candidate);
378}
379freqType wscale_from_wmax_v2(freqType & Wscale, const freqType w1, const freqType wmax, const int N) {
380 freqType Wscale_candidate;
381
382 // Version 2: quaadratic around w=0, good for w^(-2) tails
383 assert(w1 / wmax < 1./(N*N)); // if this fails, then change wmax or use Version 1
384 Wscale_candidate = w1 * wmax * N * sqrt(N*N - 1) *sqrt(wmax*wmax - N*N*w1*w1) / std::abs(wmax*wmax - w1*w1*N*N*N*N);
385
386 return std::max(Wscale, Wscale_candidate);
387}
388freqType wscale_from_wmax_v3(freqType & Wscale, const freqType w1, const freqType wmax, const int N) {
389 freqType Wscale_candidate;
390
391 // Version 3: quadratic around w=0, good for w^(-1) tails
392 Wscale_candidate = w1 * wmax * (N*N - 1*2.) / sqrt(N * (N * (wmax*wmax + w1*w1) - N*N*w1*wmax - w1*wmax));
393
394 return std::max(Wscale, Wscale_candidate);
395}

495freqType FrequencyGrid<hybridGrid>::frequency_from_t(const freqType t) const {
496 const double t_abs = std::abs(t);
497 if (t_abs < aux_pos_section_boundaries[0]) {
498 // quaddratic part
499 const double result = t_abs*t_abs / recip_curvature_quad * sgn(t);
500 assert(isfinite(result));
501 return result;
502 }
503 else if (t_abs < aux_pos_section_boundaries[1]) {
504 // linear part
505 const double result = (pos_section_boundaries[0] + (t_abs - aux_pos_section_boundaries[0]) / recip_slope_lin)* sgn(t);
506 assert(isfinite(result));
507 return result;
508 }
509 else {
510#if HYBRID_GRID_OPTION==0
511 // rational part
512 const double result = factor_rat / (1. - t_abs / rescale_rat) * sgn(t);
513 assert(isfinite(result));
514 return result;
515#else
516 // exponential part
517 const double result = (pos_section_boundaries[1] * exp((t_abs - aux_pos_section_boundaries[1]) / (recip_slope_lin * pos_section_boundaries[1]))) * sgn(t);
518 assert(isfinite(result));
519 return result;
520#endif
521 }

525freqType FrequencyGrid<hybridGrid>::t_from_frequency(const freqType w) const {
526 const double w_abs = std::abs(w);
527 if (w_abs < pos_section_boundaries[0]) {
528 // quadratic part
529 const double result = sqrt(w_abs * recip_curvature_quad) * sgn(w);
530 assert(isfinite(result));
531 return result;
532 }
533 else if (w_abs < pos_section_boundaries[1]) {
534 // linear part
535 const double result = ((w_abs - pos_section_boundaries[0]) * recip_slope_lin + aux_pos_section_boundaries[0]) * sgn(w);
536 assert(isfinite(result));
537 return result;
538 }
539 else {
540#if HYBRID_GRID_OPTION==0
541 // rational part
542 const double result = (1. - factor_rat / w_abs) * rescale_rat * sgn(w);
543 assert(isfinite(result));
544 return result;
545#else
546 // exponential part
547 const double result = sgn(w) * (aux_pos_section_boundaries[1] + log(w_abs / pos_section_boundaries[1]) * pos_section_boundaries[1] * recip_slope_lin);
548 assert(isfinite(result));
549 return result;

615 lin_fac_to_power = pow(lin_fac, power);

680double FrequencyGrid<angularGrid>::frequency_from_t(const double t) const {
681 const double i_interval = floor( (t + half_of_interval_length_for_t) * number_of_intervals / (t_upper - t_lower) );
682 const double remainder = t / half_of_interval_length_for_t - i_interval * 2; // remainder in [-1, 1]
683 //const double result = ( i_interval + 0.5 / quad_fac_recip * remainder * (std::abs(remainder) + 2.*lin_fac)) * half_of_interval_length_for_w * 2.;
684 const double result = ( i_interval + 0.5 / quad_fac_recip * sgn(remainder) * (pow(std::abs(remainder) + lin_fac, power) - lin_fac_to_power)) * half_of_interval_length_for_w * 2.;
685 assert(isfinite(result));
