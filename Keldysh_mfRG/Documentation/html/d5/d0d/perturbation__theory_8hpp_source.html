160 bareState.initialize(false); // a state with a bare vertex and a self-energy initialized at the Hartree value
161
162#if not defined(NDEBUG)
163 utils::print("Start initializing bubble object ... ", false);
164#endif
165
166 // Initialize bubble objects
167 Propagator<Q> barePropagator(bareState.Lambda, bareState.selfenergy, 'g', bareState.config); //Bare propagator
168 Propagator<Q> bareSingleScalePropagator(bareState.Lambda, bareState.selfenergy, diff ? 's' : 'g', bareState.config); //Bare propagator
169 //auto Pi = PT_initialize_Bubble(barePropagator);
170 Bubble<Q> Pi (barePropagator, bareSingleScalePropagator, diff);
171
172#if not defined(NDEBUG)
173 utils::print_add("done.", true);
174#endif
175
176 sopt_state_impl(Psi, Pi, bareState);

184template<typename Q>
185void topt_state(State<Q>& Psi) {
186
187 State<Q> bareState (Psi, Psi.Lambda);
188 bareState.initialize(false); //a state with a bare vertex and a self-energy initialized at the Hartree value
189
190 // Initialize bubble objects
191 Propagator<Q> barePropagator(Psi.Lambda, bareState.selfenergy, 'g', Psi.config); //Bare propagator
192 auto Pi = PT_initialize_Bubble(barePropagator);
193
194 State<Q> SoptPsi (Psi, Psi.Lambda);
195 //SoptPsi.initialize();
196 sopt_state_impl(SoptPsi, Pi, bareState);
197
198#if not defined(NDEBUG)
199 utils::print("Computing the vertex in TOPT ... ", false);
200#endif
201 //Calculate the bubbles -> Vertex in TOPT saved in Psi
202 Psi.vertex = SoptPsi.vertex + bareState.vertex;
203 vertexInTOPT(Psi.vertex, bareState, SoptPsi, Pi, Psi.Lambda);
204#if not defined(NDEBUG)
205 utils::print_add("done.", true);
206#endif
207
208 Psi.selfenergy = bareState.selfenergy + SoptPsi.selfenergy;
209
210}
