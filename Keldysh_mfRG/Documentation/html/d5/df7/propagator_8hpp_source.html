193template <typename Q>
194auto Propagator<Q>::GK(const double v, const int i_in) const -> Q
195{
196 if constexpr (EQUILIBRIUM and not (REG==5)) {
197 // FDT in equilibrium: (1-2*Eff_distr)*(GR-GA)
198 //return (1.-2.*Eff_distr(v))*(GR(v, i_in) - GA(v, i_in));
199 return glb_i * (Eff_fac(v,T) * 2. * myimag(GR(v, i_in))); // more efficient: only one interpolation instead of two
200 }
201 else if constexpr (EQUILIBRIUM and (REG==5)){
202 // evaluate FDT with T=Lambda
203 //return glb_i * (Eff_fac(v,Lambda) * 2. * myimag(GR(v, i_in))); // explicitly uses FDT
204 return (selfenergy.valsmooth(1, v, i_in) - glb_i * Gamma * Eff_fac(v, Lambda)) * std::norm(GR(v, i_in));
205 }
206 else {
207 // General form (Dyson equation): GR*(SigmaK+SigmaK_res)*GA
208 // Derivation of equilibrium form:
209 // \Sigma^K = (1-2n_F)(\Sigma^R-\Sigma^A), accordingly for \Sigma^K_res
210 // \Rightarrow G^K = (1-2n_F) G^R G^A [ (\Sigma+\Sigma_res)^R - (\Sigma+\Sigma_res)^A ]
211 // = (1-2n_F) G^R G^A [ (G^A)^{-1} - (G^R)^{-1} ] = (1-2n_F) (G^R-G^A)
212 // note that \Sigma_res^R = - i (Gamma+Lambda) / 2.
213 // return GR(v, i_in) * (selfenergy.valsmooth(1, v, i_in) - glb_i*(Gamma+Lambda)*(1.-2.*Eff_distr(v))) * GA(v, i_in);
214 // more efficient: only one interpolation instead of two; std::norm(c)=std::std::abs(c)^2
215 return std::norm(GR(v, i_in)) * (selfenergy.valsmooth(1, v, i_in) - glb_i * ((Gamma + Lambda) * Eff_fac(v,T)));
216 }
217}

233template <typename Q>
234auto Propagator<Q>::SK(const double v, const int i_in) const -> Q
235{
236 if constexpr (EQUILIBRIUM and not (REG==5)) {
237 // FDT in equilibrium: (1-2*Eff_distr)*(SR-SA)
238 //return (1.-2.*Eff_distr(v))*(SR(v, i_in) - myconj(SR(v, i_in)));
239 return glb_i * (Eff_fac(v,T) * 2. * myimag(SR(v, i_in)));
240 }
241 else if constexpr (EQUILIBRIUM and (REG==5)){
242 // special form of SK in the temperature flow
243 const double root_denominator = Lambda * cosh(v/(2.0*Lambda));
244 // return -glb_i * v * myimag(GR_REG5(v, i_in)) / (root_denominator*root_denominator); //explicitly uses FDT
245 return glb_i * 0.5 * Gamma * v * std::norm(GR(v, i_in)) / (root_denominator*root_denominator) ;
246 }
247 else {
248 // Derivation of general matrix form:
249 // S = - G * ( \partial_\Lambda G_0^{-1} ) * G
250 // where G = (0, G^A; G^R, G^K), G_0^{-1} = (Ginv0K, G_0^{A,-1}; G_0^{R,-1}, 0), Ginv0K = - \Sigma^K_res, \dot{Ginv0K} = i (1-2n_F)
251 // Thus, S^K = - G^R \dot{G_0^{R,-1}} G^K - G^K \dot{G_0^{A,-1}} G^A - G^R Ginv0K G^A
252 // Upon inserting G^K = (1-2n_F) (G^R-G^A), one recovers the equilibrium formula
253 //Q retarded = -0.5*glb_i*GR(v, i_in)*GK(v, i_in);
254 //Q advanced = +0.5*glb_i*GK(v, i_in)*GA(v, i_in);
255 //Q extra = -glb_i*(1.-2.*Eff_distr(v))*GR(v, i_in)*GA(v, i_in);
256 //return retarded + advanced + extra;
257 //return GK(v, i_in)*imag(GR(v, i_in)) - glb_i*(1.-2.*Eff_distr(v))*GR(v, i_in)*GA(v, i_in); // more efficient
258 //return GK(v, i_in)*imag(GR(v, i_in)) - glb_i*(Eff_fac(v)) * std::norm( GR(v, i_in) ); // more efficient
259 // most efficient: insert GK, factor out, combine real factors
260 Q gr = GR(v, i_in);
261 double gri = myimag(gr);
262 double grn = std::norm(gr);
263 return selfenergy.valsmooth(1, v, i_in) * (grn * gri) -
264 glb_i * (grn * Eff_fac(v,T) * (1. + (Gamma + Lambda) * gri));
265 }

306template <typename Q>
307auto Propagator<Q>::valsmooth(const int iK, const freqType v, const int i_in) const -> Q {
308 switch (type){
309 case 'g' : //Good ol' regular propagator
310 if constexpr (KELDYSH){
311 if constexpr(CONTOUR_BASIS != 1) {
312 switch (iK) {
313 case 0:
314 return GR(v, i_in);
315 case 1:
316 return GK(v, i_in);
317 default:
318 utils::print("ERROR! Invalid Keldysh index. Abort.");
319 assert(false);
320 }
321 }
322 else {
323 const Q GR_ = GR(v, i_in);
324 const Q GA_ = conj(GR_);
325 const Q GK_ = GK(v, i_in);
326 switch (iK){
327 case 0:
328 return 0.5*(GA_ + GR_ + GK_);
329 case 1:
330 return 0.5*(GA_ - GR_ + GK_);
331 case 2:
332 return 0.5*(-GA_ + GR_ + GK_);
333 case 3:
334 return 0.5*(-GA_ - GR_ + GK_);
335 default:
336 utils::print("ERROR! Invalid Keldysh index. Abort.");
337 assert(false);
338 }
339 }
340 }
341 else{
342 return GM(v, i_in);
343 }
344 break;
345 case 's':
346 if constexpr (KELDYSH){
347 if constexpr(CONTOUR_BASIS != 1) {
348 // Keldysh basis:
349 switch (iK){
350 case 0:
351 return SR(v, i_in);
352 case 1:
353 return SK(v, i_in);
354 default:
355 utils::print("ERROR! Invalid Keldysh index. Abort.");
356 assert(false);
357 }
358 }
359 else {
360 // Contour basis:
361 const Q SR_ = SR(v, i_in);
362 const Q SA_ = conj(SR_);
363 const Q SK_ = SK(v, i_in);
364 switch (iK){
365 case 0:
366 return 0.5*(SA_ + SR_ + SK_);
367 case 1:
368 return 0.5*(SA_ - SR_ + SK_);
369 case 2:
370 return 0.5*(-SA_ + SR_ + SK_);
371 case 3:
372 return 0.5*(-SA_ - SR_ + SK_);
373 default:
374 utils::print("ERROR! Invalid Keldysh index. Abort.");
375 assert(false);
376 }
377 }
378 }
379 else{
380 return SM(v, i_in);
381 }
382 break;
383 case 'k': // including the Katanin extension
384 if constexpr (KELDYSH){
385 if constexpr(CONTOUR_BASIS != 1) {
386 switch (iK) {
387 case 0:
388 return SR(v, i_in) + Katanin_R(v, i_in);
389 case 1:
390 return SK(v, i_in) + Katanin_K(v, i_in);
391 default:
392 utils::print("ERROR! Invalid Keldysh index. Abort.");
393 assert(false);
394 }
395 }
396 else {
397 const Q SR_ = SR(v, i_in) + Katanin_R(v, i_in);
398 const Q SA_ = conj(SR_);
399 const Q SK_ = SK(v, i_in) + Katanin_K(v, i_in);
400 switch (iK){
401 case 0: return 0.5*(SA_ + SR_ + SK_);
402 case 1: return 0.5*(SA_ - SR_ + SK_);
403 case 2: return 0.5*(-SA_ + SR_ + SK_);
404 case 3: return 0.5*(-SA_ - SR_ + SK_);
405 default:
406 utils::print("ERROR! Invalid Keldysh index. Abort.");
407 assert(false);
408 }
409 }
410 }
411 else{
412 return SM(v, i_in)
413 + GM(v, i_in) * diff_selfenergy.valsmooth(0, v, i_in) * GM(v, i_in);
414 }
415 break;
416 case 'e': // purely the Katanin extension
417 if constexpr (KELDYSH){
418 if constexpr(CONTOUR_BASIS != 1) {
419 switch (iK) {
420 case 0: return Katanin_R(v, i_in);
421 case 1: return Katanin_K(v, i_in);
422 default:
423 utils::print("ERROR! Invalid Keldysh index. Abort.");
424 assert(false);
425 }
426 }
427 else {
428 const Q SR_ = Katanin_R(v, i_in);
429 const Q SA_ = conj(SR_);
430 const Q SK_ = Katanin_K(v, i_in);
431 switch (iK){
432 case 0: return 0.5*(SA_ + SR_ + SK_);
433 case 1: return 0.5*(SA_ - SR_ + SK_);
434 case 2: return 0.5*(-SA_ + SR_ + SK_);
435 case 3: return 0.5*(-SA_ - SR_ + SK_);
436 default:
437 utils::print("ERROR! Invalid Keldysh index. Abort.");
438 assert(false);
439 }
440 }
441 }
442 else{
443 return GM(v, i_in) * diff_selfenergy.valsmooth(0, v, i_in) * GM(v, i_in);
444 }
445 break;
446 default:
447 utils::print("ERROR! Invalid Keldysh index. Abort.");
448 assert(false);
449 exit(1); // Failure
450 }
451 assert(false);
452 return 0.;
453}

588template <typename Q>
589auto Propagator<Q>::norm() const -> double {
590 double out = 0.;
591 for (int i = 0; i < nPROP; i++) {
592 if (KELDYSH) out += pow(std::abs(GR(selfenergy.Sigma.frequencies.primary_grid.get_frequency(i), 0)), 2.);
593 else out += pow(std::abs(GM(selfenergy.Sigma.frequencies.primary_grid.get_frequency(i), 0)), 2.);
594 }
595 return sqrt(out);
596}

604template <typename Q>
605auto Propagator<Q>::G0M_inv_SIAM(const freqType v, const int i_in) const -> Q {
606 if constexpr (PARTICLE_HOLE_SYMMETRY) {
607 const Q G0inv = v * ((!KELDYSH and !ZERO_T) ? M_PI*T : 1.) + Gamma * 0.5 * sign(v);
