44double Hartree_Solver::compute_Hartree_term_bracketing(const double convergence_threshold, const bool Friedel_check,
45 const bool verbose) {
46 double LHS = 1.;
47 int n_iter = 0;
48 double filling_min = 0.;
49 double filling_max = 1.;
50 while (std::abs(LHS) > convergence_threshold) {
51 assert(n_iter < 1e5); // Avoid infinite loop
52 n_iter++;
53 filling = (filling_min + filling_max) / 2.;
54 SelfEnergy<comp> Sigma_new (Lambda, config);
55 Sigma_new.initialize(config.U * filling, 0);
56 Sigma_new.Sigma.initInterpolator();
57 selfEnergy = Sigma_new;
58 const double filling_tmp = compute_filling_oneshot();
59 LHS = filling_tmp - filling;
60 if (LHS > 0.) {
61 filling_min = filling;
62 }
63 else if (LHS < 0.) {
64 filling_max = filling;
65 }
66 else { // LHS == 0.0 exactly
67 utils::print("Warning, obtained a numerically exact result for the filling. Might be suspicious.", true);
68 return config.U * filling;
69 }
70 }
71 if (verbose) {
72 utils::print_add(" ", true);
73 utils::print("Determined filling of: " + std::to_string(filling), true);
74 utils::print("Number of iterations needed: " + std::to_string(n_iter), true);

81double Hartree_Solver::compute_Hartree_term_Friedel(const double convergence_threshold) {
82 assert(ZERO_T);
83 double n_ini = 1./2.;
84 double n = n_ini;
85 double diff = 1;
86 int counter = 0;
87 while (diff > convergence_threshold) {
88 assert(counter < 1e5);
89 n_ini = n;
90 double Vg = config.U*0.5 + config.epsilon;
91 n = 1./2. - 1./M_PI * atan(Vg / Delta + config.U / Delta * (n - 1./2.));
92 diff = std::abs(n_ini - n);
93 counter++;
94 }
95 return config.U * n;
96}

104auto Hartree_Solver::operator()(const double nu) const -> double {
105 Propagator<comp> G (Lambda, selfEnergy, prop_type, config);
106 if constexpr (KELDYSH) {
107 // integrand for the filling (not the Hartree value!) given according to the formula
108 // 1/pi * n_F(nu) Im G^R(nu)
109 double val = - 1. / M_PI * fermi_distribution(nu);
110
111 if (test_different_Keldysh_component){
112 if (test_Keldysh_component == "A_real") return std::conj(G.SR(nu, 0)).real();
113 else if (test_Keldysh_component == "A_imag") return std::conj(G.SR(nu, 0)).imag();
114 else if (test_Keldysh_component == "R_real") return G.SR(nu, 0).real();
115 else if (test_Keldysh_component == "R_imag") return G.SR(nu, 0).imag();
116 else if (test_Keldysh_component == "K_real") return G.SK(nu, 0).real();
117 else if (test_Keldysh_component == "K_imag") return G.SK(nu, 0).imag();
118 else if (test_Keldysh_component == "lesser") return - 4 * fermi_distribution(nu) * G.SR(nu, 0).imag();
119 }
120 else {
121 switch (prop_type) {
122 case 'g': return val * G.GR(nu, 0).imag();
123 case 's': return val * G.SR(nu, 0).imag(); // TODO: sign and prefactor??
124 default: assert(false); return 0.;
125 }
126 }
127 assert (false);
128 return 0;
129 }
130 else { // Matsubara
135 switch (prop_type) {
136 case 'g': return (G.GM(nu, 0) - 1./(glb_i * nu * ((!KELDYSH and !ZERO_T) ? M_PI*config.T : 1.))).real();
137 case 's': return G.SM(nu, 0).real(); // TODO: sign and prefactor??
138 default: assert(false); return 0.;
139 }
140 }
