48 auto operator() (double vp) const -> double {
49 if (std::abs(vp) < v.w_upper) {
50 int index = v.get_grid_index(vp);
51 double x1 = v.get_frequency(index);
52 double x2 = v.get_frequency(index + 1);
53 if (!(x1 < x2)) {
54 index -= 1;
55 x1 = v.get_frequency(index);
56 x2 = v.get_frequency(index + 1);
57 }
58 double xd = (vp - x1) / (x2 - x1);
59
60 double f1 = Phi[iLambda * nFER * n_in + index * n_in + i_in];
61 double f2 = Phi[iLambda * nFER * n_in + (index + 1) * n_in + i_in];
62
63 return myimag((1. - xd) * f1 + xd * f2 + 2. * selfEnergy.valsmooth(0, vp, i_in));
64 }
65 else
66 return 0.;
67 }
