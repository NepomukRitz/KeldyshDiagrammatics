254template<char channel_bubble, typename I> void get_i0_left_right(const my_index_t iK, I& i0_left, I& i0_right) {
255 constexpr std::array<I, 4> Keldysh4pointdims = {2,2,2,2};
256 std::array<I ,4> alpha;
257 getMultIndex(alpha, iK, Keldysh4pointdims);
258
259 if constexpr(channel_bubble == 'a') {
260 i0_left = alpha[0]*2 + alpha[3];
261 i0_right= alpha[2]*2 + alpha[1];
262 }
263 else if constexpr(channel_bubble == 'p') {
264 i0_left = alpha[0]*2 + alpha[1];
265 i0_right= alpha[2]*2 + alpha[3];
266 }
267 else {
268 static_assert(channel_bubble=='t', "Please use a, p or t for the channels.");
269 i0_left = alpha[1]*2 + alpha[3];
270 i0_right= alpha[2]*2 + alpha[0];
271 }
272}

275template<char channel_bubble, bool is_left_vertex> auto rotate_Keldysh_matrix(const my_index_t iK) -> my_index_t {
276 constexpr std::array<my_index_t, 4> Keldysh4pointdims = {2,2,2,2};
277 std::array<my_index_t ,4> alpha;
278 getMultIndex(alpha, iK, Keldysh4pointdims);
279
280 my_index_t i0_left, i0_right;
281 if constexpr(channel_bubble == 'a') {
282
283 if constexpr(is_left_vertex) {
284 i0_left = alpha[0] * 2 + alpha[3];
285 i0_right = alpha[2] * 2 + alpha[1];
286 }
287 else {
288 i0_left = alpha[2] * 2 + alpha[1];
289 i0_right = alpha[0] * 2 + alpha[3];
290 }
291 }
292 else if constexpr(channel_bubble == 'p') {
293 if constexpr(is_left_vertex) {
294 i0_left = alpha[0]*2 + alpha[1];
295 i0_right= alpha[2]*2 + alpha[3];
296 }
297 else {
298 i0_left = alpha[2]*2 + alpha[3];
299 i0_right= alpha[0]*2 + alpha[1];
300
301 }
302 }
303 else {
304 static_assert(channel_bubble=='t', "Please use a, p or t for the channels.");
305
306 if constexpr(is_left_vertex) {
307 i0_left = alpha[3] * 2 + alpha[0];
308 i0_right = alpha[2] * 2 + alpha[1];
309 }
310 else {
311 i0_left = alpha[1] * 2 + alpha[2];
312 i0_right = alpha[0] * 2 + alpha[3];
313
314 }
315 }
316
317 const my_index_t iK_read = i0_left * 4 + i0_right;
318 assert(iK_read < 16);
319 return iK_read;
320}
