58template <typename Integrand, typename Q>
59auto Adapt<Integrand,Q>::integrate(const double a, const double b) -> Q {
60 double m, h, err_i1, err_i2, r, x[13];
61 Q i1, i2, is, f[13];
62
63 m = 0.5*(b+a); // center of the interval
64 h = 0.5*(b-a); // half width of the interval
65
66 x[0] = a; // left boundary
67 x[12] = b; // right boundary
68 f[0] = integrand(x[0]); // value at the left boundary
69 f[12] = integrand(x[12]); // value at the right boundary
70
71 for (int i=1; i<12; i++) {
72 x[i] = m + nodes[i] * h; // positions of the 13 Gauss-Kronrod nodes
73 f[i] = integrand(x[i]); // integrand values at the 13 Gauss-Kronrod nodes
74 }
75
76 // use 4-point Gauss-Lobatto rule as a first estimate
77 i1 = Gauss_Lobatto_4(h, f[0], f[4], f[8], f[12]);
78
79 // use 7-point Gauss-Kronrod rule as a second estimate
80 i2 = Gauss_Kronrod_7(h, f[0], f[2], f[4], f[6], f[8], f[10], f[12]);
81
82 // use 13-point Gauss-Kronrod rule as error estimate
83 is = Gauss_Kronrod_13(h, f[0], f[1], f[2], f[3], f[4], f[5], f[6], f[7], f[8], f[9], f[10], f[11], f[12]);
84
85 err_i1 = myabs(i1 - is); // error of first estimate compared to 13-point Gauss-Kronrod
86 err_i2 = myabs(i2 - is); // error of second estimate compared to 13-point Gauss-Kronrod
87
88 // scale tolerance: if error of i2 is smaller than error of i1 (integral is converging), increase tolerance.
89 r = (err_i1 != 0.0) ? err_i2/err_i1 : 1.0; // scaling factor r
90 tolerance_rel = (r > 0.0 && r < 1.0) ? TOL / r : TOL; // if 0 < r < 1, increase relative tolerance by factor 1/r
91
92 // if error estimate is zero, set to interval width to avoid infinite recursion
93 if (myabs(is) < 1e-20 || b <= a)
94 return myzero<Q>();
95
96 // If difference between first and second estimate and second and 13-point estimate is already smaller than absolute
97 // or relative tolerance, return second estimate, else subdivide interval.
98 // Subdivide also if the integral value is exactly zero, to avoid accidental zero result due to the choice of
99 // evaluation points.
100 if ((myabs(i2 - i1) < std::max(tolerance_abs, tolerance_rel * myabs(is))
101 && myabs(i2 - is) < std::max(tolerance_abs, tolerance_rel * myabs(is)))
102 //&& i2 != 0. // shouldn't need this safety check any more if interval is split into subintervals
103 || b-a < tolerance_abs) // do not split if the interval is very small
104 return i2;
105 else
106 return integrate(x[0], x[2], f[0], f[2], is) // subdivide interval
107 + integrate(x[2], x[4], f[2], f[4], is)
108 + integrate(x[4], x[6], f[4], f[6], is)
109 + integrate(x[6], x[8], f[6], f[8], is)
110 + integrate(x[8], x[10], f[8], f[10], is)
111 + integrate(x[10], x[12], f[10], f[12], is);
112}

114template <typename Integrand, typename Q>
115auto Adapt<Integrand,Q>::integrate(const double a, const double b, const Q fa, const Q fb, const Q is) -> Q{
116
117 double m, h, x[5];
118 Q i1, i2, f[5];
119 //double m, h, mll, ml, mr, mrr;
120 //Q fmll, fml, fm, fmr, fmrr, i2, i1;
121
122 m = 0.5*(a+b); // center of the interval
123 h = 0.5*(b-a); // half width of the interval
124
125 // positions of the Gauss-Kronrod nodes
126 x[0] = m - alpha * h;
127 x[1] = m - beta * h;
128 x[2] = m;
129 x[3] = m + beta * h;
130 x[4] = m + alpha * h;
131
132 for (int i=0; i<5; ++i)
133 f[i] = integrand(x[i]); // integrand values at the Gauss-Kronrod nodes
134
135 // first and second estimate using 4-point Gauss-Lobatto and 7-point Gauss-Kronrod
136 i1 = Gauss_Lobatto_4(h, fa, f[1], f[3], fb);
137 i2 = Gauss_Kronrod_7(h, fa, f[0], f[1], f[2], f[3], f[4], fb);
138
139 // if difference between first and second estimate is smaller than absolute or relative tolerance_rel,
140 // or nodes lie outside the interval, return second estimate, else subdivide interval
141 if (myabs(i2 - i1) < std::max(tolerance_abs, tolerance_rel * myabs(is)) || x[0] < a || b < x[4])
142 return i2;
143 else
144 return integrate(a, x[0], fa, f[0], is) // subdivide interval
145 + integrate(x[0], x[1], f[0], f[1], is)
146 + integrate(x[1], x[2], f[1], f[2], is)
147 + integrate(x[2], x[3], f[2], f[3], is)
148 + integrate(x[3], x[4], f[3], f[4], is)
149 + integrate(x[4], b, f[4], fb, is);
150}
