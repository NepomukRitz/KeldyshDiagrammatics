150template <typename Q, bool differentiated> void State<Q,differentiated>::initialize(bool checks) {
151 // Initial conditions
152 // Assign initial conditions to self energy
153 if (!KELDYSH && PARTICLE_HOLE_SYMMETRY) {
154 this->selfenergy.initialize(0., 0.);
155 }
156 else {
157 this->selfenergy.initialize(config.U / 2., 0.);
158 if (!PARTICLE_HOLE_SYMMETRY){ // SIAM in Keldysh WITHOUT particle-hole symmetry
159 assert (not PARTICLE_HOLE_SYMMETRY);
160 Hartree_Solver Hartree_Term(Lambda, config);
161 const double hartree_value = Hartree_Term.compute_Hartree_term_bracketing(1e-12, checks, checks);
162 this->selfenergy.initialize(hartree_value, 0.);
163 vertex.center_frequency_grids(shift_fgrids_woPHS ? hartree_value + config.epsilon : 0.);
164 }
165 }
166
167 // Assign initial conditions to bare vertex
168 if (KELDYSH and CONTOUR_BASIS != 1) this->vertex.initialize(-config.U/2.);
169 else this->vertex.initialize(-config.U);
170
171 this->initialized = true;
172}
